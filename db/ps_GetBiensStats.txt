ALTER PROCEDURE [dbo].[ps_GetBiensStats]
AS
BEGIN
    SELECT
        COUNT(*) as TotalBiens,
        SUM(CASE WHEN type_bien = 'Appartement' THEN 1 ELSE 0 END) as NombreAppartements,
        SUM(CASE WHEN type_bien = 'Local' THEN 1 ELSE 0 END) as NombreLocaux
    FROM TBIENS
END
